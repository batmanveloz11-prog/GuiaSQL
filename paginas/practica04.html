Modificaciones de Tabla (DDL)

1. RENAME — Renombrar la tabla


                    ALTER TABLE empleados RENAME TO colaboradores;

2. ADD COLUMN — Agregar columna


                    ALTER TABLE colaboradores ADD fecha_ingreso DATE;

3. DROP COLUMN — Eliminar columna


                    ALTER TABLE colaboradores DROP COLUMN puesto;

4. CHANGE — Renombrar columna y cambiar tipo


                    -- Renombra 'nombre' a 'nombre_completo' y ajusta su tipo
                    ALTER TABLE colaboradores CHANGE nombre nombre_completo VARCHAR(100);

5. MODIFY — Modificar tipo de dato


                    -- Cambia el tipo sin renombrar la columna
                    ALTER TABLE colaboradores MODIFY salario DECIMAL(12,2);

6. ADD CONSTRAINT PK — Agregar clave primaria


                    ALTER TABLE colaboradores
                        ADD CONSTRAINT pk_colaboradores PRIMARY KEY (id);

7. ADD CONSTRAINT FK — Agregar clave foránea


                    -- Paso 1: agregar la columna
                    ALTER TABLE colaboradores ADD id_departamento INT;

                    -- Paso 2: vincular como FK
                    ALTER TABLE colaboradores
                        ADD CONSTRAINT fk_departamento
                        FOREIGN KEY (id_departamento) REFERENCES departamentos(id);

8. ADD CONSTRAINT — UNIQUE, DEFAULT y CHECK

UNIQUE — Sin correos duplicados


                    ALTER TABLE colaboradores ADD correo VARCHAR(100);
                    ALTER TABLE colaboradores ADD CONSTRAINT uq_correo UNIQUE (correo);

DEFAULT — Salario por defecto


                    ALTER TABLE colaboradores ALTER COLUMN salario SET DEFAULT 15000.00;

CHECK — Salario siempre positivo


                    ALTER TABLE colaboradores
                        ADD CONSTRAINT chk_salario_positivo CHECK (salario > 0);

9. TRUNCATE — Vaciar la tabla


                    -- Borra todos los registros pero conserva la estructura
                    TRUNCATE TABLE colaboradores;

10. DROP TABLE — Eliminar la tabla


                    -- Elimina la tabla y toda su estructura definitivamente
                    DROP TABLE colaboradores;
