14. Manejo de Fechas y Horas

Fecha y Hora Actual


                SELECT NOW();            -- Fecha y hora actual
                SELECT CURDATE();        -- Solo fecha actual
                SELECT CURTIME();        -- Solo hora actual

UTC (Tiempo Universal Coordinado)


                SELECT UTC_DATE();       -- Fecha en UTC
                SELECT UTC_TIME();       -- Hora en UTC
                SELECT UTC_TIMESTAMP();  -- Fecha y hora en UTC

Extraer Partes de una Fecha


                SELECT YEAR(NOW());      -- Año
                SELECT MONTH(NOW());     -- Mes
                SELECT DAY(NOW());       -- Día
                SELECT HOUR(NOW());      -- Hora
                SELECT MINUTE(NOW());    -- Minutos
                SELECT SECOND(NOW());    -- Segundos

Día de la Semana


                SELECT DAYOFWEEK(NOW()); -- 1=Domingo ... 7=Sábado
                SELECT WEEKDAY(NOW());   -- 0=Lunes   ... 6=Domingo

Otras Extracciones


                SELECT DAYOFMONTH(NOW()); -- Día del mes (1-31)
                SELECT DAYOFYEAR(NOW());  -- Día del año (1-365)
                SELECT WEEK(NOW());       -- Número de semana del año
                SELECT QUARTER(NOW());    -- Trimestre (1-4)

Formatear Fechas con DATE_FORMAT


                SELECT DATE_FORMAT(NOW(), '%d/%m/%Y');  -- Ej: 16/06/2025
                SELECT DATE_FORMAT(NOW(), '%Y-%m-%d'); -- Ej: 2025-06-16

Calcular Edad con TIMESTAMPDIFF


                SELECT 
                    id, 
                    Nombre, 
                    TIMESTAMPDIFF(YEAR, FechaNacimiento, CURDATE()) AS Edad, 
                    TIMESTAMPDIFF(DAY,  FechaNacimiento, NOW())     AS Edad_Dias 
                FROM alumno 
                WHERE id = 2;

Mensaje con Años y Días Exactos


                SELECT 
                    CONCAT(
                        'Mi nombre es Gabriel y tengo ',
                        TIMESTAMPDIFF(YEAR, '2009-08-20', CURDATE()),
                        ' años y ',
                        DATEDIFF(
                            CURDATE(),
                            DATE_ADD('2009-08-20', 
                                INTERVAL TIMESTAMPDIFF(YEAR, '2009-08-20', CURDATE()) YEAR)
                        ),
                        ' días.'
                    ) AS mensaje;
