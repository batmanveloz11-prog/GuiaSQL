13. Funciones Numéricas

SUM - Suma total


                SELECT SUM(precio_compra)  AS total_compra,
                    SUM(precio_publico) AS total_publico
                FROM Producto;

AVG - Promedio


                SELECT AVG(precio_compra)  AS promedio_compra,
                    AVG(precio_publico) AS promedio_publico
                FROM Producto;

COUNT - Contar registros


                SELECT COUNT(*) AS total_productos
                FROM Producto;

MAX - Valor máximo


                SELECT MAX(precio_compra)  AS max_compra,
                    MAX(precio_publico) AS max_publico
                FROM Producto;

MIN - Valor mínimo


                SELECT MIN(precio_compra)  AS min_compra,
                    MIN(precio_publico) AS min_publico
                FROM Producto;

ABS - Valor absoluto


                SELECT nombre_producto,
                    ABS(precio_publico - precio_compra) AS diferencia
                FROM Producto;

ROUND - Redondear


                SELECT nombre_producto,
                    ROUND(precio_publico / precio_compra, 2) AS margen_ratio
                FROM Producto;

CEILING - Redondear hacia arriba


                SELECT nombre_producto,
                    CEILING(precio_compra * 1.16) AS precio_con_iva
                FROM Producto;

FLOOR - Redondear hacia abajo


                SELECT nombre_producto,
                    FLOOR(precio_publico * 0.90) AS precio_descuento
                FROM Producto;

TRUNCATE - Truncar decimales


                SELECT nombre_producto,
                    TRUNCATE(precio_publico / 3, 1) AS precio_3_msi
                FROM Producto;

SIGN - Signo del valor


                SELECT nombre_producto, stock,
                    SIGN(stock) AS hay_existencia
                FROM Producto;

Cálculos de Inventario

Valor total del inventario


                SELECT SUM(precio_compra * stock) AS valor_inventario
                FROM Producto;

Ingreso total si vendes todo


                SELECT SUM(precio_publico * stock) AS ingreso_total
                FROM Producto;

Ganancia neta si vendes todo


                SELECT SUM(precio_publico * stock) - SUM(precio_compra * stock) AS ganancia_neta
                FROM Producto;

Resumen completo del inventario


                SELECT 
                    SUM(precio_compra * stock)                                AS valor_inventario,
                    SUM(precio_publico * stock)                               AS ingreso_total,
                    SUM(precio_publico * stock) - SUM(precio_compra * stock)  AS ganancia_neta,
                    ROUND(
                        (SUM(precio_publico * stock) - SUM(precio_compra * stock))
                        / SUM(precio_compra * stock) * 100, 2
                    )                                                         AS margen_porcentaje
                FROM Producto;
