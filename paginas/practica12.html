1Convertir a Mayúsculas (UPPER)
SELECT UPPER(nombre) AS NOMBRE_MAYUSCULAS
            FROM alumno2;
2Convertir a Minúsculas (LOWER)
SELECT LOWER(ciudad)
            FROM alumno2;
Estructura e Inserción de Prueba
DESC alumno2;

SELECT * FROM alumno2;
3Eliminar Espacios en Blanco (TRIM)
SELECT TRIM(nombre) AS nombre_sin_espacios
            FROM alumno2;
4Extraer Subcadenas (SUBSTRING)
SELECT SUBSTRING(nombre, 1, 2) AS subcadena
            FROM alumno2;
5Posición de Carácter (INSTR)
SELECT email, INSTR(email, '@') AS Posicion_arroba
            FROM alumno2;
6Concatenar Cadenas (CONCAT)
SELECT CONCAT(nombre, " ", apellido) AS nombre_completo
            FROM alumno2;
7Reemplazar Caracteres (REPLACE)
SELECT REPLACE("hola", 'o', 'ofo') AS IdiomaF FROM alumno2;
SELECT REPLACE((SELECT REPLACE("hola", 'o', 'ofo') AS IdiomaF), 'a', 'afa') AS IdiomaF;
8Invertir Cadena (REVERSE)
SELECT REVERSE("Anita lava la tina");
9Longitud de Cadena (LENGTH)
SELECT nombre, LENGTH(nombre) AS LONGITUD
            FROM alumno2;
10Buscar Patrones con LIKE
SELECT nombre
            FROM alumno2
            WHERE nombre LIKE "_a%";
