. Procedimientos Almacenados

Un procedimiento almacenado (Stored Procedure) es un conjunto de instrucciones SQL agrupadas y guardadas en el servidor de base de datos que se pueden ejecutar repetidamente.

El delimitador personalizado (DELIMITER)

En MySQL, el carácter punto y coma ; es el delimitador por defecto para ejecutar instrucciones. Al crear un procedimiento, que contiene múltiples instrucciones internas terminadas en ;, es necesario cambiar temporalmente el delimitador (por ejemplo a $$ o //) para que MySQL no ejecute el código antes de terminar de definirlo.

-- Cambiar el delimitador a $$
                DELIMITER $$

                -- Cambiar el delimitador a //
                DELIMITER //
Tabla y Datos de Prueba (alumno) 
DROP TABLE IF EXISTS alumno;
CREATE TABLE alumno (
                    ID INT AUTO_INCREMENT PRIMARY KEY,
                    Nombre VARCHAR(100) NOT NULL,
                    AP VARCHAR(50) NOT NULL, 
                    AM VARCHAR(50) NOT NULL  
                )//

1Procedimiento Básico (Select All)
CREATE PROCEDURE ga()
                BEGIN
                    SELECT * FROM alumno;
                END//

Llamada al procedimiento:

CALL ga()//
2Seleccionar por ID (Select ID)
CREATE PROCEDURE selecionar(IN p_id INT)
                BEGIN
                    SELECT ID, Nombre, AP, AM 
                    FROM alumno 
                    WHERE ID = p_id;
                END//

Llamada al procedimiento:

CALL selecionar(3)//
3Insertar Datos (Insert)
CREATE PROCEDURE insertar(
                    IN p_nombre VARCHAR(100),
                    IN p_ap VARCHAR(100),
                    IN p_am VARCHAR(100)
                )
                BEGIN
                    INSERT INTO alumno (Nombre, AP, AM) 
                    VALUES (p_nombre, p_ap, p_am);
                END//

Llamada al procedimiento:

CALL insertar('Gabriel', 'Aguirre', 'Edomex')//
4Actualizar Datos (Update)
CREATE PROCEDURE actualizar(
                    IN p_id INT,
                    IN p_nombre VARCHAR(100),
                    IN p_ap VARCHAR(100),
                    IN p_am VARCHAR(100)
                )
                BEGIN
                    UPDATE alumno 
                    SET Nombre = p_nombre, 
                        AP = p_ap, 
                        AM = p_am 
                    WHERE ID = p_id;
                END//

Llamada al procedimiento:

CALL actualizar(2, 'Pedro Modificado', 'Perez', 'Cracio')//
5Eliminar Datos (Delete)
CREATE PROCEDURE eliminar(IN p_id INT)
                BEGIN
                    DELETE FROM alumno 
                    WHERE ID = p_id;
                END//

Llamada al procedimiento:

CALL eliminar(5)//
6Ver y Eliminar Procedimientos

Ver todos los procedimientos en la base de datos actual:

SHOW PROCEDURE STATUS 
                WHERE Db = DATABASE()//

Eliminar un procedimiento existente:

DROP PROCEDURE IF EXISTS nombre_del_procedimiento//
